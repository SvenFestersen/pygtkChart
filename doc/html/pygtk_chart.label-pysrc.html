681      text_width_n, text_height_n = layout.get_pixel_size() 
682      text_width = text_width_n * abs(math.cos(angle)) + text_height_n * abs(math.sin(angle)) 
683      text_height = text_height_n * abs(math.cos(angle)) + text_width_n * abs(math.sin(angle)) 
684      height_delta = text_height - text_height_n 
685      x, y = pos 
686      ref = (0, -text_height) 
687      if anchor == ANCHOR_TOP_LEFT: 
688          ref = (0, 0) 
689      elif anchor == ANCHOR_TOP_RIGHT: 
690          ref = (-text_width, height_delta) 
691      elif anchor == ANCHOR_BOTTOM_RIGHT: 
692          ref = (-text_width, -text_height) 
693      elif anchor == ANCHOR_CENTER: 
694          ref = (-text_width / 2, -text_height / 2) 
695      elif anchor == ANCHOR_TOP_CENTER: 
696          ref = (-text_width / 2, 0) 
697      elif anchor == ANCHOR_BOTTOM_CENTER: 
698          ref = (-text_width / 2, -text_height) 
699      elif anchor == ANCHOR_LEFT_CENTER: 
700          ref = (0, -text_height / 2) 
701      elif anchor == ANCHOR_RIGHT_CENTER: 
702          ref = (-text_width, -text_height / 2) 
703      x += ref[0] 
704      y += ref[1] 
705      return x, y
